def detect_outliers_iqr(data):
    data = sorted(data)
    q1 = np.percentile(data, 25)
    q3 = np.percentile(data, 75)
    # print(q1, q3)
    IQR = q3-q1
    lowerbound = q1-(1.5*IQR)
    upperbound = q3+(1.5*IQR)
    
    for i in data: 
        if (i<lowerbound or i>upperbound):
            outliers.append(i)
    return outliers# Driver code
